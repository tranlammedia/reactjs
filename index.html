Ví dụ 2

            function Button({ title, href, onClick }) {
                let Component = 'button'
                const props = {}
    
                if (onClick) {
                    props.onClick = onClick
                }
    
                if (href) {
                    Component = 'a'
                    props.href = href
                }
                return (
                    <Component {...props}>{title} </Component>
                )
            }
    
            // App.js
            function App() {
                return (
                    <Button 
                        title="Click me"
                        href="https://fullstack.edu.vn/"
                        // onClick={() => console.log(Math.random())}
                    />
                )    
            }
